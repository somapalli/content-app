-- =============================================================================
-- TRANSACTION MONITORING MODEL - ORACLE DDL + SAMPLE DATA
-- ABC Bank | Payment Processing System
-- =============================================================================
-- SCHEMA OVERVIEW
--   1.  LOOKUP TABLES          (static reference data)
--   2.  CORE TABLES            (transaction, event, allocation)
--   3.  MONITORING TABLES      (materialized view table, exceptions)
--   4.  CONSTRAINTS & INDEXES
--   5.  SAMPLE DATA
--       Scenario A: Normal Auth → Incremental → Capture → Settle
--       Scenario B: Auth → Partial Void → Capture → Settle
--       Scenario C: Auth → Capture → Settle → Refund
--       Scenario D: Auth → Capture → Settle → Chargeback → Chargeback Won
--       Scenario E: Monthly Fee Applied
--       Scenario F: Positive Adjustment (manual credit)
--       Scenario G: Duplicate Auth Attempt (exception raised)
--       Scenario H: Capture Exceeds Authorization (exception raised)
-- =============================================================================


-- -----------------------------------------------------------------------------
-- SECTION 1: LOOKUP TABLES
-- -----------------------------------------------------------------------------

CREATE TABLE lkp_transaction_type (
    type_id         NUMBER(4)       GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    type_code       VARCHAR2(30)    NOT NULL,
    type_name       VARCHAR2(100)   NOT NULL,
    description     VARCHAR2(500),
    is_active       NUMBER(1)       DEFAULT 1 NOT NULL CHECK (is_active IN (0,1)),
    created_at      TIMESTAMP       DEFAULT SYSTIMESTAMP NOT NULL,
    CONSTRAINT uq_txn_type_code UNIQUE (type_code)
);

CREATE TABLE lkp_transaction_subtype (
    subtype_id      NUMBER(4)       GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    type_id         NUMBER(4)       NOT NULL,   -- parent type
    subtype_code    VARCHAR2(30)    NOT NULL,
    subtype_name    VARCHAR2(100)   NOT NULL,
    description     VARCHAR2(500),
    is_active       NUMBER(1)       DEFAULT 1 NOT NULL CHECK (is_active IN (0,1)),
    created_at      TIMESTAMP       DEFAULT SYSTIMESTAMP NOT NULL,
    CONSTRAINT uq_txn_subtype_code UNIQUE (subtype_code),
    CONSTRAINT fk_subtype_type FOREIGN KEY (type_id)
        REFERENCES lkp_transaction_type (type_id)
);

CREATE TABLE lkp_transaction_status (
    status_id       NUMBER(4)       GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    type_id         NUMBER(4)       NOT NULL,   -- status scoped to a type
    status_code     VARCHAR2(30)    NOT NULL,
    status_name     VARCHAR2(100)   NOT NULL,
    is_terminal     NUMBER(1)       DEFAULT 0 NOT NULL CHECK (is_terminal IN (0,1)),  -- no further transitions allowed
    is_active       NUMBER(1)       DEFAULT 1 NOT NULL CHECK (is_active IN (0,1)),
    created_at      TIMESTAMP       DEFAULT SYSTIMESTAMP NOT NULL,
    CONSTRAINT uq_txn_status_code UNIQUE (status_code),
    CONSTRAINT fk_status_type FOREIGN KEY (type_id)
        REFERENCES lkp_transaction_type (type_id)
);

CREATE TABLE lkp_event_type (
    event_type_id   NUMBER(4)       GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    event_type_code VARCHAR2(30)    NOT NULL,
    event_type_name VARCHAR2(100)   NOT NULL,
    affects_balance NUMBER(1)       DEFAULT 1 NOT NULL CHECK (affects_balance IN (0,1)),
    description     VARCHAR2(500),
    is_active       NUMBER(1)       DEFAULT 1 NOT NULL CHECK (is_active IN (0,1)),
    created_at      TIMESTAMP       DEFAULT SYSTIMESTAMP NOT NULL,
    CONSTRAINT uq_evt_type_code UNIQUE (event_type_code)
);

-- Enforces only valid type/subtype combos can be used on a transaction
CREATE TABLE lkp_valid_type_subtype (
    type_id         NUMBER(4)       NOT NULL,
    subtype_id      NUMBER(4)       NOT NULL,
    is_active       NUMBER(1)       DEFAULT 1 NOT NULL CHECK (is_active IN (0,1)),
    CONSTRAINT pk_valid_type_subtype PRIMARY KEY (type_id, subtype_id),
    CONSTRAINT fk_vts_type    FOREIGN KEY (type_id)    REFERENCES lkp_transaction_type    (type_id),
    CONSTRAINT fk_vts_subtype FOREIGN KEY (subtype_id) REFERENCES lkp_transaction_subtype (subtype_id)
);

-- Allowed status transitions per transaction type  
-- e.g., Credit Card: AUTHORIZED → CAPTURED → SETTLED
CREATE TABLE lkp_status_transition (
    transition_id       NUMBER(6)       GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    type_id             NUMBER(4)       NOT NULL,
    from_status_id      NUMBER(4),      -- NULL = initial state (no prior status)
    to_status_id        NUMBER(4)       NOT NULL,
    is_active           NUMBER(1)       DEFAULT 1 NOT NULL CHECK (is_active IN (0,1)),
    CONSTRAINT uq_status_transition UNIQUE (type_id, from_status_id, to_status_id),
    CONSTRAINT fk_st_type   FOREIGN KEY (type_id)        REFERENCES lkp_transaction_type   (type_id),
    CONSTRAINT fk_st_from   FOREIGN KEY (from_status_id) REFERENCES lkp_transaction_status (status_id),
    CONSTRAINT fk_st_to     FOREIGN KEY (to_status_id)   REFERENCES lkp_transaction_status (status_id)
);


-- -----------------------------------------------------------------------------
-- SECTION 2: CORE TABLES
-- -----------------------------------------------------------------------------

CREATE TABLE merchant (
    merchant_id         VARCHAR2(20)    PRIMARY KEY,
    merchant_name       VARCHAR2(200)   NOT NULL,
    merchant_category   VARCHAR2(10),   -- MCC code
    acquiring_bank_id   VARCHAR2(20),
    is_active           NUMBER(1)       DEFAULT 1 NOT NULL CHECK (is_active IN (0,1)),
    created_at          TIMESTAMP       DEFAULT SYSTIMESTAMP NOT NULL
);

CREATE TABLE merchant_account (
    account_id          VARCHAR2(20)    PRIMARY KEY,
    merchant_id         VARCHAR2(20)    NOT NULL,
    account_type        VARCHAR2(20)    NOT NULL,  -- CHECKING, SETTLEMENT, FEE_COLLECTION
    currency_code       CHAR(3)         DEFAULT 'USD' NOT NULL,
    is_active           NUMBER(1)       DEFAULT 1 NOT NULL CHECK (is_active IN (0,1)),
    created_at          TIMESTAMP       DEFAULT SYSTIMESTAMP NOT NULL,
    CONSTRAINT fk_macct_merchant FOREIGN KEY (merchant_id) REFERENCES merchant (merchant_id)
);

-- ── TRANSACTION ──────────────────────────────────────────────────────────────
CREATE TABLE txn (
    transaction_id          VARCHAR2(36)    PRIMARY KEY,    -- UUID
    merchant_id             VARCHAR2(20)    NOT NULL,
    account_id              VARCHAR2(20)    NOT NULL,
    terminal_id             VARCHAR2(50),
    type_id                 NUMBER(4)       NOT NULL,
    subtype_id              NUMBER(4)       NOT NULL,
    status_id               NUMBER(4)       NOT NULL,

    -- Three-amount model: critical for reconciliation
    requested_amount        NUMBER(18,2)    NOT NULL,
    approved_amount         NUMBER(18,2),
    settled_amount          NUMBER(18,2),
    currency_code           CHAR(3)         DEFAULT 'USD' NOT NULL,

    -- Timestamps
    initiated_at            TIMESTAMP       NOT NULL,
    processed_at            TIMESTAMP,
    settled_at              TIMESTAMP,

    -- Dedup / external refs
    idempotency_key         VARCHAR2(100),
    external_ref_id         VARCHAR2(100),           -- processor / network reference

    -- Hierarchy: refund or chargeback links back to original
    parent_transaction_id   VARCHAR2(36),

    -- Card / channel context
    entry_mode              VARCHAR2(30),             -- CARD_PRESENT, CNP, CONTACTLESS, TOKEN
    channel                 VARCHAR2(30),             -- POS, ECOMMERCE, MOTO, API
    auth_code               VARCHAR2(20),
    network_response_code   VARCHAR2(10),             -- "00" approved, "05" declined, etc.
    card_bin                VARCHAR2(8),              -- first 6-8 digits for BIN analysis
    card_last4              VARCHAR2(4),
    card_network            VARCHAR2(20),             -- VISA, MASTERCARD, AMEX, DISCOVER

    -- Risk
    risk_score              NUMBER(5,2),
    risk_decision           VARCHAR2(20),             -- APPROVED, REVIEWED, DECLINED

    -- Audit
    created_by              VARCHAR2(100)   DEFAULT 'SYSTEM' NOT NULL,
    source_system           VARCHAR2(50),
    created_at              TIMESTAMP       DEFAULT SYSTIMESTAMP NOT NULL,
    updated_at              TIMESTAMP       DEFAULT SYSTIMESTAMP NOT NULL,

    CONSTRAINT fk_txn_merchant  FOREIGN KEY (merchant_id)  REFERENCES merchant          (merchant_id),
    CONSTRAINT fk_txn_account   FOREIGN KEY (account_id)   REFERENCES merchant_account  (account_id),
    CONSTRAINT fk_txn_type      FOREIGN KEY (type_id)      REFERENCES lkp_transaction_type    (type_id),
    CONSTRAINT fk_txn_subtype   FOREIGN KEY (subtype_id)   REFERENCES lkp_transaction_subtype (subtype_id),
    CONSTRAINT fk_txn_status    FOREIGN KEY (status_id)    REFERENCES lkp_transaction_status  (status_id),
    CONSTRAINT fk_txn_parent    FOREIGN KEY (parent_transaction_id) REFERENCES txn (transaction_id),
    CONSTRAINT fk_txn_vts       FOREIGN KEY (type_id, subtype_id)
                                REFERENCES lkp_valid_type_subtype (type_id, subtype_id)
);

-- ── EVENT ─────────────────────────────────────────────────────────────────────
CREATE TABLE txn_event (
    event_id                VARCHAR2(36)    PRIMARY KEY,    -- UUID
    transaction_id          VARCHAR2(36)    NOT NULL,
    event_type_id           NUMBER(4)       NOT NULL,
    sequence_number         NUMBER(6)       NOT NULL,       -- ordered within transaction

    amount                  NUMBER(18,2)    NOT NULL,       -- signed: negative for reversals
    currency_code           CHAR(3)         DEFAULT 'USD' NOT NULL,

    event_status            VARCHAR2(20)    DEFAULT 'COMPLETED' NOT NULL,
                                                            -- PENDING, COMPLETED, FAILED, REVERSED

    -- Who/what triggered this event
    initiated_by            VARCHAR2(20)    NOT NULL,       -- SYSTEM, MERCHANT, NETWORK, OPERATOR

    -- External timestamps vs internal receipt
    network_timestamp       TIMESTAMP,
    received_at             TIMESTAMP       NOT NULL,
    processed_at            TIMESTAMP,

    -- Processor reference
    processor_event_id      VARCHAR2(100),

    -- Reversal chain: self-referencing
    is_reversal             NUMBER(1)       DEFAULT 0 NOT NULL CHECK (is_reversal IN (0,1)),
    reversed_event_id       VARCHAR2(36),                   -- FK to the event being reversed

    -- Extensible metadata (network-specific fields, raw response, etc.)
    metadata                CLOB            CHECK (metadata IS NULL OR metadata = metadata),  -- valid CLOB

    created_at              TIMESTAMP       DEFAULT SYSTIMESTAMP NOT NULL,

    CONSTRAINT fk_evt_txn           FOREIGN KEY (transaction_id)   REFERENCES txn          (transaction_id),
    CONSTRAINT fk_evt_type          FOREIGN KEY (event_type_id)    REFERENCES lkp_event_type (event_type_id),
    CONSTRAINT fk_evt_reversal      FOREIGN KEY (reversed_event_id) REFERENCES txn_event    (event_id),
    CONSTRAINT uq_evt_seq           UNIQUE (transaction_id, sequence_number)  -- no duplicate sequence within a txn
);

-- ── ALLOCATION (MONEY MOVEMENT) ───────────────────────────────────────────────
-- Each allocation_group_id groups the lines of a single money movement.
-- Double-entry invariant: SUM(amount) per group = 0.
CREATE TABLE txn_allocation (
    allocation_id           NUMBER(12)      GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    allocation_group_id     VARCHAR2(36)    NOT NULL,       -- UUID — groups SOURCE + TARGET lines
    event_id                VARCHAR2(36)    NOT NULL,
    role                    VARCHAR2(10)    NOT NULL CHECK (role IN ('SOURCE','TARGET')),
    flow_type               VARCHAR2(30)    NOT NULL,       -- AUTH, INCREMENTAL, CAPTURE, VOID, SETTLEMENT, REFUND, etc.
    amount                  NUMBER(18,2)    NOT NULL,       -- signed: negative for SOURCE, positive for TARGET
    ledger_entry_id         VARCHAR2(36),                   -- FK to double-entry ledger (future)
    allocated_at            TIMESTAMP       DEFAULT SYSTIMESTAMP NOT NULL,

    CONSTRAINT fk_alloc_event FOREIGN KEY (event_id) REFERENCES txn_event (event_id)
);


-- -----------------------------------------------------------------------------
-- SECTION 3: MONITORING TABLES
-- -----------------------------------------------------------------------------

-- Pre-aggregated monitoring record per transaction (CQRS read side)
-- Refreshed by a background job or trigger after each event is posted
CREATE TABLE txn_monitor (
    transaction_id              VARCHAR2(36)    PRIMARY KEY,
    merchant_id                 VARCHAR2(20)    NOT NULL,
    current_status_code         VARCHAR2(30)    NOT NULL,

    -- Running totals
    total_authorized            NUMBER(18,2)    DEFAULT 0,
    total_captured              NUMBER(18,2)    DEFAULT 0,
    total_settled               NUMBER(18,2)    DEFAULT 0,
    total_refunded              NUMBER(18,2)    DEFAULT 0,
    total_chargedback           NUMBER(18,2)    DEFAULT 0,
    net_position                NUMBER(18,2)    DEFAULT 0,  -- authorized - voided - refunded - chargedback

    -- Flags
    has_open_chargeback         NUMBER(1)       DEFAULT 0 NOT NULL CHECK (has_open_chargeback IN (0,1)),
    is_reconciled               NUMBER(1)       DEFAULT 0 NOT NULL CHECK (is_reconciled IN (0,1)),
    capture_exceeds_auth        NUMBER(1)       DEFAULT 0 NOT NULL CHECK (capture_exceeds_auth IN (0,1)),

    -- Reconciliation
    reconciliation_batch_id     VARCHAR2(50),

    -- Event tracking
    event_count                 NUMBER(6)       DEFAULT 0,
    last_event_type_code        VARCHAR2(30),
    last_event_at               TIMESTAMP,

    -- Timing SLA monitoring
    auth_to_capture_minutes     NUMBER(10,2),   -- latency between auth and capture
    capture_to_settle_minutes   NUMBER(10,2),   -- latency between capture and settlement
    processing_lag_minutes      NUMBER(10,2),   -- total initiated_at to settled_at

    -- Anomaly flags as JSON array: ["CAPTURE_EXCEEDS_AUTH","LATE_SETTLEMENT"]
    anomaly_flags               VARCHAR2(1000),

    updated_at                  TIMESTAMP       DEFAULT SYSTIMESTAMP NOT NULL,

    CONSTRAINT fk_monitor_txn      FOREIGN KEY (transaction_id) REFERENCES txn      (transaction_id),
    CONSTRAINT fk_monitor_merchant FOREIGN KEY (merchant_id)    REFERENCES merchant  (merchant_id)
);

-- Structured exception log — replaces ad-hoc status flags
CREATE TABLE txn_exception (
    exception_id            NUMBER(12)      GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    transaction_id          VARCHAR2(36)    NOT NULL,
    event_id                VARCHAR2(36),               -- nullable: may be transaction-level
    exception_code          VARCHAR2(50)    NOT NULL,   -- CAPTURE_EXCEEDS_AUTH, DUPLICATE_AUTH, STALE_AUTHORIZATION, etc.
    exception_severity      VARCHAR2(20)    NOT NULL    CHECK (exception_severity IN ('INFO','WARNING','CRITICAL')),
    exception_message       VARCHAR2(1000),
    detected_at             TIMESTAMP       DEFAULT SYSTIMESTAMP NOT NULL,
    resolved_at             TIMESTAMP,
    resolved_by             VARCHAR2(100),
    resolution_notes        VARCHAR2(1000),
    auto_resolvable         NUMBER(1)       DEFAULT 0 NOT NULL CHECK (auto_resolvable IN (0,1)),
    is_resolved             NUMBER(1)       DEFAULT 0 NOT NULL CHECK (is_resolved IN (0,1)),

    CONSTRAINT fk_exc_txn   FOREIGN KEY (transaction_id) REFERENCES txn       (transaction_id),
    CONSTRAINT fk_exc_event FOREIGN KEY (event_id)       REFERENCES txn_event (event_id)
);

-- Transaction lookup patterns
CREATE INDEX idx_txn_merchant_id        ON txn (merchant_id);
CREATE INDEX idx_txn_status             ON txn (status_id);
CREATE INDEX idx_txn_initiated_at       ON txn (initiated_at);
CREATE INDEX idx_txn_external_ref       ON txn (external_ref_id);
CREATE INDEX idx_txn_idempotency        ON txn (idempotency_key);
CREATE INDEX idx_txn_parent             ON txn (parent_transaction_id);
CREATE INDEX idx_txn_card_bin           ON txn (card_bin);
CREATE INDEX idx_txn_merchant_date      ON txn (merchant_id, initiated_at);

-- Event lookup patterns
CREATE INDEX idx_evt_transaction_id     ON txn_event (transaction_id);
CREATE INDEX idx_evt_type_id            ON txn_event (event_type_id);
CREATE INDEX idx_evt_received_at        ON txn_event (received_at);
CREATE INDEX idx_evt_txn_seq            ON txn_event (transaction_id, sequence_number);

-- Allocation
CREATE INDEX idx_alloc_group_id         ON txn_allocation (allocation_group_id);
CREATE INDEX idx_alloc_event_id         ON txn_allocation (event_id);
CREATE INDEX idx_alloc_flow_type        ON txn_allocation (flow_type);

-- Monitoring
CREATE INDEX idx_monitor_merchant       ON txn_monitor (merchant_id);
CREATE INDEX idx_monitor_status         ON txn_monitor (current_status_code);
CREATE INDEX idx_monitor_chargeback     ON txn_monitor (has_open_chargeback);

-- Exceptions
CREATE INDEX idx_exc_transaction_id     ON txn_exception (transaction_id);
CREATE INDEX idx_exc_code               ON txn_exception (exception_code);
CREATE INDEX idx_exc_severity           ON txn_exception (exception_severity);
CREATE INDEX idx_exc_unresolved         ON txn_exception (is_resolved, detected_at);

-- 1. Full event timeline for any transaction
SELECT
    e.sequence_number,
    et.event_type_code,
    e.amount,
    e.event_status,
    e.is_reversal,
    e.network_timestamp,
    e.initiated_by
FROM txn_event e
JOIN lkp_event_type et ON e.event_type_id = et.event_type_id
WHERE e.transaction_id = 'T-A001'
ORDER BY e.sequence_number;

-- 2. Allocation net-zero check per group (should = 0 for all healthy transactions)
SELECT
    allocation_group_id,
    flow_type,
    SUM(amount)  AS net_balance,
    CASE WHEN SUM(amount) = 0 THEN 'BALANCED' ELSE 'IMBALANCED' END AS balance_check
FROM txn_allocation
GROUP BY allocation_group_id, flow_type
ORDER BY allocation_group_id;

-- 3. Open exceptions dashboard
SELECT
    e.exception_code,
    e.exception_severity,
    t.merchant_id,
    m.merchant_name,
    e.exception_message,
    e.detected_at
FROM txn_exception e
JOIN txn t ON e.transaction_id = t.transaction_id
JOIN merchant m ON t.merchant_id = m.merchant_id
WHERE e.is_resolved = 0
ORDER BY e.exception_severity DESC, e.detected_at;

-- 4. Transaction monitoring dashboard — all merchants
SELECT
    m.merchant_name,
    mon.current_status_code,
    mon.total_authorized,
    mon.total_settled,
    mon.total_refunded,
    mon.total_chargedback,
    mon.net_position,
    mon.capture_exceeds_auth,
    mon.event_count,
    mon.anomaly_flags
FROM txn_monitor mon
JOIN merchant m ON mon.merchant_id = m.merchant_id
ORDER BY m.merchant_name, mon.last_event_at DESC;

-- 5. Chargeback exposure by merchant
SELECT
    m.merchant_name,
    COUNT(*)            AS chargeback_count,
    SUM(mon.total_chargedback) AS total_exposure
FROM txn_monitor mon
JOIN merchant m ON mon.merchant_id = m.merchant_id
WHERE mon.total_chargedback > 0
GROUP BY m.merchant_name
ORDER BY total_exposure DESC;

-- 6. Transactions with unbalanced allocations (fraud/data quality signal)
SELECT
    a.allocation_group_id,
    t.transaction_id,
    m.merchant_name,
    SUM(a.amount) AS imbalance_amount
FROM txn_allocation a
JOIN txn_event e ON a.event_id = e.event_id
JOIN txn t ON e.transaction_id = t.transaction_id
JOIN merchant m ON t.merchant_id = m.merchant_id
GROUP BY a.allocation_group_id, t.transaction_id, m.merchant_name
HAVING SUM(a.amount) <> 0
ORDER BY ABS(SUM(a.amount)) DESC;

-- 7. Parent-child refund/chargeback chain view
SELECT
    p.transaction_id    AS original_txn,
    p.requested_amount  AS original_amount,
    c.transaction_id    AS child_txn,
    cst.subtype_name    AS child_type,
    c.requested_amount  AS child_amount,
    c.initiated_at      AS child_initiated
FROM txn p
JOIN txn c  ON c.parent_transaction_id = p.transaction_id
JOIN lkp_transaction_subtype cst ON c.subtype_id = cst.subtype_id
ORDER BY p.transaction_id, c.initiated_at;
